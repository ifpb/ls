                                  Resposta:

                                  • array.map(e => 2 \* e)

                                  • array.filter(e => !(e & 1))
                                  • array.filter((e, i) => !(i & 1))
                                  • array.reduce((s, v) => s + v)<br />array.reduce((s, v) => s \* v)
                                  • array.filter(e => !(i & 1)).reduce((s, v) => s + v)
                                  • array.filter(e => !(i & 1)).map(e => 2 \* e).reduce((s, v) => s + v)
